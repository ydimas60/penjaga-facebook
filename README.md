-- Cara Install --

$ pkg update

$ pkg install php

$ pkg install git

$ pkg upgrade

$ git clone https://github.com/ngotakdikit/penjaga-facebook.git

-- Cara Menjalankan --

$ cd penjaga-facebook

$ php fb.php
